172 void envelopPoint(const T x, const T y)
173 {
174 min.x = std::min(x, min.x);

176 min.y = std::min(y, min.y);
177 max.y = std::max(y, max.y);

182template <
183 typename T,
184 typename TVertexIter,
185 typename TGetVertexCoordX,
186 typename TGetVertexCoordY>
187Box2d<T> envelopBox(
188 TVertexIter first,
189 TVertexIter last,
190 TGetVertexCoordX getX,
191 TGetVertexCoordY getY)
192{
193 const T max = std::numeric_limits<T>::max();
194 Box2d<T> box = {{max, max}, {-max, -max}};
195 for(; first != last; ++first)
196 {
197 box.envelopPoint(getX(*first), getY(*first));
198 }
199 return box;
200}

256struct CDT_EXPORT Triangle

258 VerticesArr3 vertices;
259 NeighborsArr3 neighbors;
260
266 static Triangle
267 make(const array<VertInd, 3>& vertices, const array<TriInd, 3>& neighbors)
268 {
269 Triangle t = {vertices, neighbors};
270 return t;
271 }
272
275 std::pair<TriInd, VertInd> next(const VertInd i) const
276 {
277 assert(vertices[0] == i || vertices[1] == i || vertices[2] == i);
278 if(vertices[0] == i)
279 {
280 return std::make_pair(neighbors[0], vertices[1]);
281 }
282 if(vertices[1] == i)
283 {
284 return std::make_pair(neighbors[1], vertices[2]);
285 }
286 return std::make_pair(neighbors[2], vertices[0]);
287 }
290 std::pair<TriInd, VertInd> prev(const VertInd i) const
291 {
292 assert(vertices[0] == i || vertices[1] == i || vertices[2] == i);
293 if(vertices[0] == i)
294 return std::make_pair(neighbors[2], vertices[2]);
295 if(vertices[1] == i)
296 return std::make_pair(neighbors[0], vertices[0]);
297 return std::make_pair(neighbors[1], vertices[1]);
298 }
299
300 bool containsVertex(const VertInd i) const
301 {
302 return std::find(vertices.begin(), vertices.end(), i) != vertices.end();
303 }
304};

469template <>
470struct hash<CDT::Edge>
471{
473 std::size_t operator()(const CDT::Edge& e) const
474 {
475 return hashEdge(e);
476 }
477
478private:
479 static void hashCombine(std::size_t& seed, const CDT::VertInd& key)
480 {
481#ifdef CDT_CXX11_IS_SUPPORTED
482 typedef std::hash<CDT::VertInd> Hasher;
483#else
484 typedef boost::hash<CDT::VertInd> Hasher;
485#endif
486 seed ^= Hasher()(key) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
487 }
488 static std::size_t hashEdge(const CDT::Edge& e)
489 {
490 const std::pair<CDT::VertInd, CDT::VertInd>& vv = e.verts();
491 std::size_t seed1(0);
492 hashCombine(seed1, vv.first);
493 hashCombine(seed1, vv.second);
494 std::size_t seed2(0);
495 hashCombine(seed2, vv.second);
496 hashCombine(seed2, vv.first);
497 return std::min(seed1, seed2);
498 }
499};
